from flask import Flask, jsonify
from flask_cors import CORS

app = Flask(__name__)
CORS(app)

# Define a static list of data
data = [{'name': 'Joe', 'age': 17}, {'name': 'Jane', 'age': 25}, {'name': 'Bob', 'age': 40}]

@app.route('/data')
def get_data():
    return jsonify(data)

if __name__ == '__main__':
    app.run()
